\begin{itemize}
	\item File Structure for XAMPP
		\begin{itemize}
			\item \verb|%XAMPP location%/htdocs/php/|
		\end{itemize}
	\item Hello.php
		\begin{verbatim}
			<html>
			<head>
			</head>
			    <body>
			        <p>Hello world</p>
			    </body>
			</html>
		\end{verbatim}
		PHP code should be placed between ''\verb|<?php|'' and ''\verb|?>|'' lines like so:
		\begin{verbatim}
			<html>
			<head>
			</head>
			    <body>
			        <p>Hello world</p>

			        <?php
			            echo "<p>PHP text</p>";
			        ?>
			    </body>
			</html>
		\end{verbatim}
	\item Running file
		\begin{itemize}
			\item Type \verb|localhost\php\hello.php| in the address bar of your web browser.
			\item If you view the source file of the web page there will be no PHP block, only the HTML paragraph that was echo'ed
		\end{itemize}
\end{itemize}
